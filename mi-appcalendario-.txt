    <script>
        // Variables globales de estado
        let activeTab = 'fundamentos';
        let currentTurno = 'manana'; // 'manana' o 'tarde'
        let selectedDay = 1;

        // Estructura de un mes (28 días para patrón cíclico limpio de 4 semanas)
        // Días de entrenamiento: Martes (2), Jueves (4), Sábado (6), Domingo (7)

        const daysData = {
            1: { name: "Lunes", desc: "Descanso y Estudio Liviano", type: "rest" },
            2: { name: "Martes", desc: "Calistenia (Fuerza Torso) + Estudio", type: "mixed" },
            3: { name: "Miércoles", desc: "Descanso Total / Enfoque Proyecto", type: "rest" },
            4: { name: "Jueves", desc: "Calistenia + Trote Intervalos + Estudio", type: "mixed" },
            5: { name: "Viernes", desc: "Descanso / Preparación de Tirada", type: "rest" },
            6: { name: "Sábado", desc: "Tirada Larga (Objetivo 21K) + Estudio", type: "mixed" },
            7: { name: "Domingo", desc: "Calistenia Suave + Trote Recuperativo", type: "mixed" }
        };

        // Guardar estado del checklist en LocalStorage para no perder progreso
        let userProgress = JSON.parse(localStorage.getItem('neuro_progress')) || {};

        // Cambiar entre pestañas
        function switchTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(el => el.classList.add('hidden'));
            document.getElementById(`content-${tabId}`).classList.remove('hidden');

            document.querySelectorAll('[id^="tab-"]').forEach(btn => {
                btn.className = "px-4 py-2 text-sm font-semibold rounded-lg shrink-0 transition-all text-gray-400 hover:text-white";
            });
            const activeBtn = document.getElementById(`tab-${tabId}`);
            if (activeBtn) {
                activeBtn.className = "px-4 py-2 text-sm font-semibold rounded-lg shrink-0 transition-all bg-indigo-500/10 text-indigo-400 border border-indigo-500/20";
            }
            activeTab = tabId;
        }

        // Cambiar turno de trabajo
        function toggleTurno(turno) {
            currentTurno = turno;
            const btnManana = document.getElementById('btnTurnoManana');
            const btnTarde = document.getElementById('btnTurnoTarde');

            if (turno === 'manana') {
                btnManana.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-bold rounded-lg transition-all duration-300 bg-emerald-500 text-gray-950 shadow-lg shadow-emerald-500/20";
                btnTarde.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-bold rounded-lg transition-all duration-300 text-gray-400 hover:text-white";
            } else {
                btnTarde.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-bold rounded-lg transition-all duration-300 bg-indigo-500 text-gray-950 shadow-lg shadow-indigo-500/20";
                btnManana.className = "flex-1 sm:flex-initial px-4 py-2 text-xs font-bold rounded-lg transition-all duration-300 text-gray-400 hover:text-white";
            }
            // Recargar vista actual de checklist
            renderChecklist();
        }

        // Inicializar el calendario de 28 días
        function renderCalendar() {
            const grid = document.getElementById('calendarGrid');
            grid.innerHTML = '';

            // Encabezados de días de la semana
            const weekDays = ["Lun", "Mar", "Mié", "Jue", "Vie", "Sáb", "Dom"];
            weekDays.forEach(wd => {
                const header = document.createElement('div');
                header.className = "text-center text-xs font-bold text-gray-500 py-1";
                header.innerText = wd;
                grid.appendChild(header);
            });

            // Generar 28 días (4 semanas completas)
            for (let i = 1; i <= 28; i++) {
                const dayNum = i;
                const weekdayIndex = ((i - 1) % 7) + 1;
                const dayInfo = daysData[weekdayIndex];

                const dayBtn = document.createElement('button');
                dayBtn.id = `cal-day-${dayNum}`;
                dayBtn.onclick = () => selectDay(dayNum);

                // Determinar clases de color base según el tipo de día
                let colorClasses = "bg-gray-950 border-gray-800 hover:border-gray-700";
                if (dayInfo.type === 'mixed') {
                    colorClasses = "bg-emerald-950/20 border-emerald-500/20 text-emerald-300 hover:border-emerald-500/40";
                } else if (dayInfo.type === 'study_only') {
                    colorClasses = "bg-indigo-950/20 border-indigo-500/20 text-indigo-300 hover:border-indigo-500/40";
                }

                dayBtn.className = `h-12 flex flex-col items-center justify-between p-1.5 rounded-lg border text-xs transition-all ${colorClasses}`;
                
                // Mostrar número de día y un punto de completado si aplica
                const isSelected = selectedDay === dayNum ? 'ring-2 ring-indigo-500' : '';
                dayBtn.className += ` ${isSelected}`;

                dayBtn.innerHTML = `
                    <span class="font-bold block text-left w-full">${dayNum}</span>
                    <span id="dot-${dayNum}" class="w-1.5 h-1.5 rounded-full bg-transparent"></span>
                `;

                grid.appendChild(dayBtn);
                updateDayCompletedDot(dayNum);
            }
        }

        function selectDay(dayNum) {
            // Deseleccionar anterior
            const prevSelected = document.getElementById(`cal-day-${selectedDay}`);
            if (prevSelected) prevSelected.classList.remove('ring-2', 'ring-indigo-500');

            selectedDay = dayNum;
            
            // Seleccionar nuevo
            const newSelected = document.getElementById(`cal-day-${dayNum}`);
            if (newSelected) newSelected.classList.add('ring-2', 'ring-indigo-500');

            const weekdayIndex = ((dayNum - 1) % 7) + 1;
            const dayInfo = daysData[weekdayIndex];

            document.getElementById('selectedDayLabel').innerText = `${dayNum} (${dayInfo.name})`;
            document.getElementById('selectedDayDesc').innerText = dayInfo.desc;

            renderChecklist();
        }

        // Generar las tareas específicas de un día considerando la pauta y el turno de trabajo
        function getTasksForDay(dayNum) {
            const weekdayIndex = ((dayNum - 1) % 7) + 1;
            const dayInfo = daysData[weekdayIndex];
            const tasks = [];

            // Suplementos fijos diarios
            tasks.push({ id: 'creatina', label: "5g de Creatina post-entrenamiento (o con agua)", time: "Todo el día" });
            tasks.push({ id: 'fosfatidilserina', label: "300mg de Fosfatidilserina con comida principal", time: "Mediodía" });

            // Pauta de entrenamientos y nootrópicos específicos de días activos
            if (weekdayIndex === 2) { // Martes: Fuerza
                tasks.push({ id: 'tirosina', label: "1000mg L-Tirosina en ayunas antes de estudiar", time: "Antes de Estudio" });
                tasks.push({ id: 'alfagpc', label: "600mg Alfa GPC antes de estudiar", time: "Antes de Estudio" });
                tasks.push({ id: 'huperzina', label: "200mcg Huperzina A antes de estudiar (Ciclo Activo)", time: "Antes de Estudio" });
                tasks.push({ id: 'estudio', label: "Estudio intensivo y código del proyecto (2 horas)", time: "Bloque Cognitivo" });
                tasks.push({ id: 'calistenia', label: "Entrenamiento Calistenia: Dominadas/Fondos/Flexiones (RIR 2)", time: "Entrenamiento" });
            } 
            else if (weekdayIndex === 4) { // Jueves: Calistenia + Intervalos
                tasks.push({ id: 'tirosina', label: "1000mg L-Tirosina en ayunas antes de estudiar", time: "Antes de Estudio" });
                tasks.push({ id: 'alfagpc', label: "600mg Alfa GPC antes de estudiar", time: "Antes de Estudio" });
                tasks.push({ id: 'huperzina', label: "200mcg Huperzina A antes de estudiar (Ciclo Activo)", time: "Antes de Estudio" });
                tasks.push({ id: 'estudio', label: "Estudio intensivo y desarrollo de proyecto (2 horas)", time: "Bloque Cognitivo" });
                tasks.push({ id: 'calistenia_rapida', label: "Calistenia rápida (30 min): Superseries de volumen", time: "Entrenamiento" });
                tasks.push({ id: 'intervalos', label: "Carrera: Intervalos de Umbral (10 min suave + 4x3 min rápido + 5 min suave)", time: "Entrenamiento" });
            }
            else if (weekdayIndex === 6) { // Sábado: Tirada Larga
                tasks.push({ id: 'tirosina', label: "1000mg L-Tirosina en ayunas antes de estudiar", time: "Antes de Estudio" });
                tasks.push({ id: 'alfagpc', label: "600mg Alfa GPC antes de estudiar", time: "Antes de Estudio" });
                tasks.push({ id: 'huperzina', label: "200mcg Huperzina A antes de estudiar (Ciclo Activo)", time: "Antes de Estudio" });
                tasks.push({ id: 'estudio', label: "Estudio y revisión de proyecto (1.5 horas)", time: "Bloque Cognitivo" });
                tasks.push({ id: 'tiradalarga', label: "Carrera Larga: 10K+ a ritmo de 6:00/km (Ruta plana en Samil)", time: "Entrenamiento" });
            }
            else if (weekdayIndex === 7) { // Domingo: Activo Suave (Huperzina descansa)
                tasks.push({ id: 'calistenia_suave', label: "Calistenia Técnica: Movilidad y estiramientos", time: "Entrenamiento" });
                tasks.push({ id: 'trote_recup', label: "Trote de Recuperación Activa: 20 min sumamente suave", time: "Entrenamiento" });
                tasks.push({ id: 'plan_semana', label: "Planificar calendario y comidas para turnos rotativos", time: "Higiene Mental" });
            }
            else {
                // Días de descanso: Lunes, Miércoles, Viernes
                tasks.push({ id: 'estudio_ligero', label: "Estudio suave o lectura técnica de desarrollo (45 min)", time: "Bloque Cognitivo" });
                tasks.push({ id: 'descanso_activo', label: "Estiramientos suaves / Caminar 30 mins", time: "Recuperación" });
            }

            // Alerta de sueño basada en el turno
            if (currentTurno === 'manana') {
                tasks.push({ id: 'sueno_manana', label: "Higiene de Sueño: Acostarse antes de las 22:30. Luz apagada y sin pantallas.", time: "Noche" });

        // Renderizar el checklist del día seleccionado adaptado a los turnos de Stellantis
        function renderChecklist() {
            const container = document.getElementById('checklistContainer');
            container.innerHTML = '';

            const tasks = getTasksForDay(selectedDay);

            // Ordenar las tareas cronológicamente dependiendo del Turno de Stellantis
            const sortedTasks = sortTasksByShift(tasks);

            sortedTasks.forEach(task => {
                const isChecked = userProgress[`day-${selectedDay}-${task.id}`] ? 'checked' : '';
                const textStyle = isChecked ? 'line-through text-gray-500' : 'text-gray-200';
                
                const taskDiv = document.createElement('div');
                taskDiv.className = `flex items-start gap-3 p-3 bg-gray-950/60 hover:bg-gray-950 rounded-xl border border-gray-800 transition-colors`;
                
                taskDiv.innerHTML = `
                    <input type="checkbox" id="task-${task.id}" ${isChecked} 
                        onclick="toggleTask('${task.id}')" 
                        class="w-4 h-4 mt-1 rounded border-gray-700 text-indigo-500 focus:ring-indigo-500 bg-gray-900 cursor-pointer">
                    <div class="flex-1">
                        <label for="task-${task.id}" class="text-xs font-medium cursor-pointer ${textStyle}">
                            ${task.label}
                        </label>
                        <span class="block text-[10px] text-gray-500 mt-1 uppercase font-bold tracking-wider">${task.time}</span>
                    </div>
                `;
                container.appendChild(taskDiv);
            });
        }

        // Ordenar las tareas dependiendo de la estrategia diaria del turno rotativo de Stellantis
        function sortTasksByShift(tasks) {
            // El orden cronológico cambia según Stellantis sea mañana o tarde
            const orderManana = ["Antes de Estudio", "Bloque Cognitivo", "Mediodía", "Entrenamiento", "Todo el día", "Noche", "Higiene Mental", "Recuperación"];
            const orderTarde = ["Antes de Estudio", "Bloque Cognitivo", "Entrenamiento", "Todo el día", "Mediodía", "Noche", "Higiene Mental", "Recuperación"];

            const activeOrder = currentTurno === 'manana' ? orderManana : orderTarde;

            return [...tasks].sort((a, b) => {
                return activeOrder.indexOf(a.time) - activeOrder.indexOf(b.time);
            });
        }

        // Marcar/Desmarcar tarea
        function toggleTask(taskId) {
            const key = `day-${selectedDay}-${taskId}`;
            if (userProgress[key]) {
                delete userProgress[key];
            } else {
                userProgress[key] = true;
            }
            localStorage.setItem('neuro_progress', JSON.stringify(userProgress));
            
            renderChecklist();
            updateDayCompletedDot(selectedDay);
        }

        // Actualizar el indicador visual de completado en la cuadrícula del calendario
        function updateDayCompletedDot(dayNum) {
            const dot = document.getElementById(`dot-${dayNum}`);
            if (!dot) return;

            const dayTasks = getTasksForDay(dayNum);
            const completedCount = dayTasks.filter(task => userProgress[`day-${dayNum}-${task.id}`]).length;

            if (completedCount === dayTasks.length && dayTasks.length > 0) {
                dot.className = "w-1.5 h-1.5 rounded-full bg-emerald-400 shadow shadow-emerald-400/50";
            } else if (completedCount > 0) {
                dot.className = "w-1.5 h-1.5 rounded-full bg-indigo-400";
            } else {
                dot.className = "w-1.5 h-1.5 rounded-full bg-transparent";
            }
        }

        function resetDayProgress() {
            const dayTasks = getTasksForDay(selectedDay);
            dayTasks.forEach(task => {
                delete userProgress[`day-${selectedDay}-${task.id}`];
            });
            localStorage.setItem('neuro_progress', JSON.stringify(userProgress));
            renderChecklist();
            updateDayCompletedDot(selectedDay);
        }

        // Ejecutar funciones iniciales al cargar el documento
        window.onload = () => {
            renderCalendar();
            selectDay(1);
        };
    </script>
